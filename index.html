window.addEventListener("DOMContentLoaded", () => { const emberContainer = document.getElementById("embers"); if (!emberContainer) { console.log("❌ #embers not found in HTML"); return; } function createEmber() { const ember = document.createElement("div"); ember.classList.add("ember"); ember.style.left = Math.random() * 100 + "vw"; const size = Math.random() * 4 + 2; ember.style.width = size + "px"; ember.style.height = size + "px"; ember.style.animationDuration = (3 + Math.random() * 5) + "s"; emberContainer.appendChild(ember); setTimeout(() => { ember.remove(); }, 8000); } setInterval(createEmber, 120); });
